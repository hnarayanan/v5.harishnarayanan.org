  server_name v5.harishnarayanan.org;

  listen *:443 ssl http2;

  access_log /home/ubuntu/sites/v5.harishnarayanan.org/logs/nginx/access.log;
  error_log /home/ubuntu/sites/v5.harishnarayanan.org/logs/nginx/error.log;

  charset utf-8;
  client_max_body_size 75M;

  ssl_certificate /etc/letsencrypt/live/v1.harishnarayanan.org/fullchain.pem;

  add_header X-Content-Type-Options nosniff;
  add_header X-XSS-Protection "1; mode=block";
